// Draw a rotating cube import { Hopper } from "../js/renderer.build.js"; try { const hopper = Hopper({ canvasID: "render-output", enableDebug: false, debugRefreshDelay: 1000, forceWebGL1: false, antialias: true, stencil: true, powerPreference: "high-performance", failIfMajorPerformanceCaveat: true }); const Common = hopper.Common; const Vec2 = hopper.Vec2; const Vec3 = hopper.Vec3; const Vec4 = hopper.Vec4; const Quat = hopper.Quat; const Transform = hopper.Transform; const Mat4 = hopper.Mat4; hopper.enableDepth(); hopper.enableCulling(); hopper.clearColor(Vec4(0.2, 0.3, 0.3, 1.0)); const prog = hopper.createBasicTextureProgram(); const cube = hopper.createCube(prog); const texture2DLoader = hopper.createTexture2DLoader(); const texture = await texture2DLoader.loadCheckerboard({ brightColor: Vec4(200, 200, 200, 255), darkColor: Vec4(50, 50, 50, 255), minf: "nearest", magf: "nearest", wraps: "repeat", wrapt: "repeat", mipmap: false, anisotropy: false }); prog.use(); let rotY = 0; hopper.render(dt => { rotY += dt * 0.001; hopper.resize(); hopper.clear(); const projection = Transform.perspective(Common.toRadians(90.0), hopper.getAspect(), 0.1, 100.0); const view = Transform.lookAt(Vec3(0.0, 0.0, 2.0), Vec3(0.0, 0.0, 0.0), Vec3(0.0, 1.0, 0.0)); const model = Transform.fromRotationTranslationScale(Quat().setAxisAngle(Vec3(1, 1, 1).norm(), rotY), Vec3(0.0, 0.0, 0.0), Vec3(1.0, 1.0, 1.0)); prog.updateProjection(projection); prog.updateModelView(view.mul(model)); prog.updateColor(Vec4((Math.cos(rotY) + 1.0) * 0.5, (Math.sin(rotY) + 1.0) * 0.5, 0.5, 1.0)); prog.updateTexTransform(Vec2(0.0, 0.0)); prog.updateTexMultiplier(Vec2(2.0, 2.0)); prog.setTexture(texture); cube.bind(); cube.draw(); }); } catch (e) { const errorOutput = document.getElementById("error-output"); errorOutput.innerHTML = e.toString(); errorOutput.style.display = "block"; }
